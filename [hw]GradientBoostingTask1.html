class GradientBoost():
    def __init__(self, N):
        self.weights = []
        self.models = []
        self.N = N


    def fit(self, X, y):
        y_ = np.array(y)

        for i in range(self.N):
            regr = DecisionTreeRegressor(max_depth=3)
            regr.fit(X, y_)
            self.models.append(regr)
            self.weights.append(0.9)
            y_ = y - self.predict(X)

        return self


    def predict(self, X):
        r = np.zeros(X.shape[0])

        for i in range(len(self.models)):
            r += self.models[i].predict(X) * self.weights[i]

        return r > 0.5
